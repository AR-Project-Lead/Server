Table building {
  name int [PK]
  accessibility bool
}

Table room {
  room_number varchar
  floor int
  building_name varchar [ref: < building.name]
  coordinate "float[][2]"
  room_type room.room_type
  tags "varchar[]"
  accessibility bool
  inside_accessibliity bool
  accessible_door "int[]" [ref: > room_door.id]
  min_stairs_needed int [Note: 'When accessibility is false']

  indexes {
    (room_number, floor, building_name) [PK]
  }
}

Enum room.room_type {
  classroom
  office
  hallway
  bathroom
  staircase
  lecture_hall
  elevator
  closet
}

Table door {
  id varchar
  location varchar
  building_name varchar [ref: < building.name]
  accessibility bool
  wheelchair_button bool
  coordinate "float[]"

  indexes {
    (id, building_name) [PK]
  }

  Note: 'We could find a better way to identify the doors later (building door)'
}

Table room_door {
  id int
  room_number varchar [ref: - room.room_number]
  floor int 
  building_name varchar
  accessible_building_door varchar [ref: <> door.id]

  indexes {
    (id, room_number, floor, building_name) [PK]
  }
}
